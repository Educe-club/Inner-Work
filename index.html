    <script>

        const videos = {

            fitness: [

                { title: "Top Charts", url: "https://www.youtube.com/embed/09R8_2nJtjg", tags: ["Music", "Pop"], date: "2024-02-18" }

            ],

            tech: [

                { title: "Tech Talk 1", url: "https://www.youtube.com/embed/dQw4w9WgXcQ", tags: ["Technology", "AI"], date: "2024-02-23" },

                { title: "AI and Future", url: "https://www.youtube.com/embed/3JZ_D3ELwOQ", tags: ["Technology", "Future"], date: "2024-02-15" }

            ]

        };

        

        function loadVideos(topic) {

            history.pushState({ topic }, "", `?topic=${topic}`); // Updates the URL dynamically

            displayVideos(videos[topic]);

        }

        

        function displayVideos(videoArray) {

            const videoList = document.getElementById("videoList");

            videoList.innerHTML = "";

            

            videoArray.sort((a, b) => new Date(b.date) - new Date(a.date));

            

            videoArray.forEach(video => {

                let videoItem = document.createElement("div");

                videoItem.classList.add("video-item");

                

                let title = document.createElement("div");

                title.classList.add("video-title");

                title.innerText = video.title;

                

                let iframe = document.createElement("iframe");

                iframe.src = video.url;

                iframe.allow = "accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture";

                iframe.allowFullscreen = true;

                

                let tags = document.createElement("div");

                tags.classList.add("video-tags");

                tags.innerText = "Tags: " + video.tags.join(", ");

                

                videoItem.appendChild(title);

                videoItem.appendChild(iframe);

                videoItem.appendChild(tags);

                videoList.appendChild(videoItem);

            });

        }

        

        function searchVideos() {

            let query = document.getElementById("searchInput").value.toLowerCase();

            let filteredVideos = [];

            for (let category in videos) {

                filteredVideos = filteredVideos.concat(videos[category].filter(video => 

                    video.title.toLowerCase().includes(query) || 

                    video.tags.some(tag => tag.toLowerCase().includes(query))

                ));

            }

            displayVideos(filteredVideos);

        }

        

        function displayTopics() {

            const topicList = document.getElementById("topicList");

            topicList.innerHTML = "";

            let topics = Object.keys(videos).sort();

            topics.forEach(topic => {

                let button = document.createElement("button");

                button.innerText = topic.charAt(0).toUpperCase() + topic.slice(1);

                button.onclick = () => loadVideos(topic);

                topicList.appendChild(button);

            });

        }


        window.onload = function() {

            displayTopics();

            const urlParams = new URLSearchParams(window.location.search);

            const topic = urlParams.get("topic");


            if (topic && videos[topic]) {

                loadVideos(topic);

            } else {

                let allVideos = Object.values(videos).flat();

                displayVideos(allVideos);

            }

        };
